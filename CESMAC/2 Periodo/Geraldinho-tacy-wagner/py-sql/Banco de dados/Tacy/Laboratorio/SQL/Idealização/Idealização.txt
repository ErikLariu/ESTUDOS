----------------------------------------------------

Resumo de COMMAND LINES:

(DML) - Comandos de Manipulação de Dados

INSERT - INSERIR 
INSERT INTO - INSERIR DENTRO DE *
UPDATE - ATUALIZA UMA LISTA DE REGISTROS
DELETE - DELETA REGISTROS
VALUES - VALORES

(DDL) - Comandos sados para criar, modificar e excluir REGISTROS

CREATE TABELE - CRIAR TABELA 
ALTER TABELE - ALTERAR TABELA   
DROP TABELE - EXCLUI UMA TABELA 

(OPC) - COMMAND LINES DDL - Opcionais

PRIMARY KEY - CHAVE DE LISTA PRIMEARIA (UNICA E NÃO REPETITTIVA)
FOREIGN KEY -CHAVE ESTRANGEIRA (FK)
SERIAL - (AUTO INCREMENT da PRIMARY KEY) (POSTGRE)
OR 
AUTO_INCREMENT (SQL)
REFERENCE - REFERENCIA UM REGISTRO

(OUTROS) 

WHERE - CONDIÇÃO 
NOT NULL - NÃO PODE ESTAR VAZIO

(TIPOS DE DADOS)

INT - INTEIRO
BIGINT - GRANDE NUMERO INTEIRO
SMALLINT - PEQUENO NUMERO INTEIRO
DECIMAL - NUMERO DECIAL
FLOAT - NUMERO NÃO INTEIRO (PONTO FLUTUANTE)
REAL - (PONTO FLUTUANTE)

DATE - DATA (XX/XX/XX)
TIME - HORARIO (XX:XX:XX)
DATATIME - DATA E HORARIO (XX/XX/XX, XX:XX:XX)

CHAR - TEXTO DEFINIDO (texto lido pelo N de linhas DEFINIDAS)
VARCHAR - TEXTO FORMADO (texto lido pelo N de linhas TOTAIS)
TEXT - TEXTO (texto longo)

(RESTRICÕES)

UNIQUE - duas linhas na tabela não podem ter o mesmo valor 

(Praticas "lEGAIS")
